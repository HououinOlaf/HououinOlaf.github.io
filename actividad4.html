# 01_OBJETIVO_DEL_DESPLIEGUE

El siguiente conjunto de instrucciones representa una configuración básica de Hardening para un servidor Linux. El objetivo es minimizar la superficie de ataque controlando estrictamente el tráfico entrante y saliente.

  • > Política por Defecto: Se aplica un enfoque de "Denegar todo" (DROP) para garantizar que solo pase lo explícitamente permitido.
  • > Stateful Inspection: Se utilizan módulos de estado para permitir respuestas a conexiones ya establecidas.
  • > Servicios Críticos: Se habilitan puertos específicos para DNS, SMTP (Correo) y HTTP (Web).

# 02_SECUENCIA_DE_COMANDOS

root@servidor: /etc/iptables
# 1. Establecer política restrictiva por defecto (Bloquear entrada) root@server:~# iptables -P INPUT DROP
root@servidor: /etc/iptables
# 2. Permitir reenvío de paquetes de conexiones ya establecidas root@server:~# iptables -A FORWARD -m state --state ESTABLISHED,RELATED -j ACCEPT
root@servidor: /etc/iptables
# 3. Permitir tráfico de salida DNS (TCP Puerto 53) root@server:~# iptables -A OUTPUT -p tcp --dport 53 -m state --state NEW -j ACCEPT
root@servidor: /etc/iptables
# 4. Permitir tráfico SMTP entrante (Correo - Puerto 25) root@server:~# iptables -A INPUT -p tcp --dport 25 -m state --state NEW -j ACCEPT

# 5. Permitir tráfico SMTP saliente (Correo - Puerto 25) root@server:~# iptables -A OUTPUT -p tcp --dport 25 -m state --state NEW -j ACCEPT
root@servidor: /etc/iptables
# 6. Permitir tráfico HTTP Web entrante (Puerto 80) root@server:~# iptables -A INPUT -p tcp --dport 80 -m state --state NEW -j ACCEPT

# 7. Regla redundante/adicional para HTTP (Según reporte) root@server:~# iptables -A INPUT -p tcp --dport 80 -m state --state NEW -j ACCEPT
