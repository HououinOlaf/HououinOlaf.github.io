# 03_ANATOMÍA_Y_ANÁLISIS

3. Anatomía de un comando iptables:

root@kali: ~
root@kali:~# iptables -A INPUT -p tcp -m multiport --dports 80,443 -j ACCEPT

4. Este comando permite:

> REF: iptables -A INPUT -p tcp ... --dports 80,443 ...

"Permite la entrada de tráfico a través del Protocolo TCP hacia los puertos 80 y 443."

# 05_ANÁLISIS_DE_REGLA_6

6. ¿Qué hace esta regla?

root@kali: ~
root@kali:~# iptables -A INPUT -i eth0 -p tcp -m multiport --dports 22,80,443 \
-m state --state NEW,ESTABLISHED -j ACCEPT

// RESPUESTA:

  • > Añade una regla a la cadena de entrada INPUT.
  • > Para la red eth0.
  • > Permite Protocolos TCP dirigidos a los puertos 22 (SSH), 80 y 443.
  • > Cuando el estado sea Nuevo o Ya establecido.
  • > Y la acción Final es Aceptar (ACCEPT) los Paquetes.

# 06_ESCRITURA_DE_REGLAS_(7-11)

root@kali: ~ (bash)
# 7. Permitir tráfico HTTP entrante root@kali:~# iptables -A INPUT -p tcp --dport 80 -j ACCEPT
root@kali: ~ (bash)
# 8. Permitir todo el tráfico saliente root@kali:~# iptables -A OUTPUT -j ACCEPT
root@kali: ~ (bash)
# 9. Permitir SSH solo desde la IP 192.168.1.50 root@kali:~# iptables -A INPUT -p tcp -s 192.168.1.50 --dport 22 -j ACCEPT
root@kali: ~ (bash)
# 10. Permitir tráfico TCP (80, 443) solo si es ESTABLISHED/RELATED root@kali:~# iptables -A INPUT -p tcp -m multiport --dports 80,443 -m state --state ESTABLISHED,RELATED -j ACCEPT
root@kali: ~ (bash)
# 11. Permitir tráfico TCP por eth0 a 22, 80, 443 (NEW y ESTABLISHED) root@kali:~# iptables -A INPUT -i eth0 -p tcp -m multiport --dports 22,80,443 -m state --state NEW,ESTABLISHED -j ACCEPT
